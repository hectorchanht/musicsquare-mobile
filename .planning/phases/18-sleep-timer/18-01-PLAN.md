sleep-timer.ts exports (all pure, no DOM mutation beyond canFadeVolume's restored probe write):
```typescript
export function computeDeadline(now: number, minutes: number): number;            // now + minutes*60_000
export function isExpired(now: number, deadline: number | null): boolean;          // deadline != null && now >= deadline
export function remainingMs(now: number, deadline: number | null): number;         // deadline==null ? 0 : max(0, deadline-now)
export function fadeVolumeAt(elapsed: number, totalMs: number, startVol: number): number; // clamp [0,1], totalMs<=0 -> 0
export function canFadeVolume(audio: { volume: number }): boolean;                  // write-then-readback feature detect
export type SleepMode = 'off' | 'minutes' | 'end-of-track';
export type RepeatMode = 'off' | 'one';
export function decideEndedAction(sleepMode: SleepMode, repeatMode: RepeatMode): 'sleep-stop' | 'repeat-rewind' | 'advance';
```

sleepTimer.svelte.ts exports:
```typescript
class SleepTimer {
  mode: 'off' | 'minutes' | 'end-of-track';   // $state
  deadline: number | null;                      // $state — ABSOLUTE Date.now()+ms (D-14)
  selectedMinutes: number | null;               // $state — for the active-duration highlight (D-11)
  remaining: number;                            // $state — ms, driven by a 1s UI tick while active
  get active(): boolean;                        // mode !== 'off'
  set(mode: 'minutes', minutes: number): void;
  set(mode: 'end-of-track'): void;
  restart(): void;                              // re-set from selectedMinutes, FRESH deadline (D-11)
  cancel(): void;                               // mode='off', deadline=null, clear tick (D-09 silent)
}
export const sleepTimer: SleepTimer;
```
